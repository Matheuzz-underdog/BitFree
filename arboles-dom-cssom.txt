3. FLUJO BACKEND (MVC)
=============================================================

app.js
  └── routes/views.routes.js  →  GET /, /que-es-linux, /encuesta, /timeline
  │                               → res.render("{pagina}.ejs")
  │
  └── routes/encuesta.route.js  →  GET  /api/encuesta
  │                               →  GET  /api/encuesta/distros
  │                               →  POST /api/encuesta
  │                                  │
  │                                  └── controllers/encuesta.controller.js
  │                                      ├── validaciones inline (email, sexo, edad, nombreCompleto, sanitizar)
  │                                      └── delegacion a:
  │                                          │
  │                                          └── models/encuesta.model.js
  │                                              ├── crear(data)       → INSERT INTO encuesta_linux
  │                                              ├── obtenerTodas()    → SELECT *
  │                                              ├── buscarPorCorreo() → SELECT WHERE correo = ?
  │                                              └── obtenerDistros()  → SELECT distro
  │                                                  │
  │                                                  └── connections/connection.js
  │                                                      └── mysql.createConnection({
  │                                                            host: 'localhost',
  │                                                            user: 'root',
  │                                                            password: '',
  │                                                            database: 'encuesta'
  │                                                          })
  │
  └── public/javascripts/encuesta-conection.js  →  fetch("/api/encuesta/...")
                                                    POST → submit form
                                                    GET /distros → cargarTop3()


4. ESQUEMA BASE DE DATOS
=============================================================

Base: encuesta
Tabla: encuesta_linux

CREATE TABLE encuesta_linux (
  id               INT AUTO_INCREMENT PRIMARY KEY,
  nombre_completo  VARCHAR(255) NOT NULL,
  edad             INT DEFAULT NULL,
  sexo             ENUM('Masculino','Femenino','Otro') NOT NULL,
  correo           VARCHAR(255) NOT NULL UNIQUE,
  distro           VARCHAR(255) NOT NULL
);
